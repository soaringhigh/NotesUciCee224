				{\strut Example~\theexmp};}}
	}%
	{\mdfsetup{%
			frametitle={%
				\tikz[baseline=(current bounding box.east),outer sep=0pt]
				\node[anchor=east,rectangle,fill=cyan!30]
				{\strut Example~\theexmp:~#1};}}%
	}%

		\label{#2}}{\end{mdframed}}
